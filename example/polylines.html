polylines
e.g:

var elem = document.getElementById("warp");

var mapname = new gmap({

target:elem,

center:"39.9388838,116.3974589",//default in beijing

zoom:7

});

mapname.onMapLoad = function(){

//划线

mapname.polylines({

path: ["39.9388838,116.3974589","40.9388838,117.3974589","40.9338838,115.39712389"],

strokeColor: '#FF0000',

strokeWeight: 3

})

//标点

mapname.setMarker([{

position:"39.9388838,116.3974589"

},{

position:"40.9388838,117.3974589"

},{

position:"40.9338838,115.39712389"

}])

}

Properties

path

Type: Array [stringLaglnt]

The initial Map center. Required.
geodesic

Type: boolean

是否开启
strokeColor

Type: RGB

线颜色支持css3颜色
strokeOpacity

Type: number

0~1
strokeWeight

Type: number

线粗度
map

Type: mapObject

地图名称(不填写默认当前)

Method

getPolyLines

Type: object

clearPolylines
移除划线
