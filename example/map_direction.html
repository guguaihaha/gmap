direction
e.g:

var elem = document.getElementById("warp");

var mapname = new gmap({

target:elem,

center:"39.9388838,116.3974589",

zoom:12

});

mapname.onMapLoad = function(){

mapname.setMarker([{

position:"39.9388838,116.3974589"

},{

position:"39.1388838,116.3974589"

}])

mapname.direction({

origin:"39.9388838,116.3974589",//required

destination:"39.1388838,116.3974589",//required

travelMode:"driving",//required

suppressMarkers:true,

strokeColor:"#194D7B",

success:function(array){

array.each(function(n){

//每个n是个对象

/**

* {

duration:<string>,

distance:<string>,

step:<array>,

route:<object>,

type:<string>

* }

* **/

})

}

});

}

Properties

origin

Type: String [LatLng]

起点. Required.
destination

Type: String [LatLng]

到达终点. Required.
travelMode

Type: String [LatLng]

使用那种交通方式?driving,walking,bicycling,transit
avoidFerries

Type: boolean

如果设置为真,则避免使用独轮(可选)
avoidHighways

Type: boolean

如果为真,则避免使用高铁(可选)
durationInTraffic

Type: boolean

次选项开启后实施提供交通路况,只能针对支持地图设备使用(可选)
optimizeWaypoints

Type: boolean

开启可选经过点(可选)
provideRouteAlternatives

Type: boolean

是否提供路由选择(可选)
unitSystem

Type: UnitSystem

返回距离使用单位,目前是"KM"(可选) 目前2中方式:
google.maps.UnitSystem.METRIC
google.maps.UnitSystem.IMPERIAL
waypoints

Type: Array

数组,导航信息经过的点(可选)
suppressMarkers

Type: boolean

是否显示mark(可选)
strokeColor

Type: string

导航颜色,支持rgb与css3(可选)
success(routeArray)

type:function

导航完毕后调用的接口入口(可选) 返回到达目的地几种线路数组, 每个数组内包含一下对象
duration,<string>到达目的地的用时
distance,<string>到达目的地的距离
step,<array>具体的导航细节描述数组
route<object>导航路有对象
type:<string>该导航的交通方式

Method

getDirectionDisplay

return Array;

绘画地图导航路线对象(线路地图类)

getDirectionsService

获取当前导航服务对象(文字类)

getDirectionResponse

根据directionService返回的response对象
